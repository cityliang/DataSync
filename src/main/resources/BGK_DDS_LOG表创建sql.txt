create table BGK_DDS_LOG
(
  ID               VARCHAR2(32) not null,
  SENDCODE          VARCHAR2(32),
  DDS_DATE        VARCHAR2(10),
  DDS_NAME        VARCHAR2(100),
  CLIENT_SERVER   VARCHAR2(32),
  USER_ID         VARCHAR2(32),
  TOTAL_COUNT     NUMBER(8),
  OK_COUNT        NUMBER(8),
  BEGIN_TIME      DATE,
  END_TIME        DATE,
  MEMO            VARCHAR2(200)
)
tablespace HZXGHIS
  pctfree 10
  initrans 1
  maxtrans 255
  storage
  (
    initial 128K
    minextents 1
    maxextents unlimited
  );
-- Add comments to the table 
comment on table BGK_DDS_LOG
  is '日志表';
-- Add comments to the columns 
comment on column BGK_DDS_LOG.id
  is '唯一编码';
comment on column BGK_DDS_LOG.SENDCODE
  is '发送编码';
comment on column BGK_DDS_LOG.DDS_DATE
  is '同步日期条件yyyymmdd';
comment on column BGK_DDS_LOG.DDS_NAME
is '同步名称（table/view名称）';
comment on column BGK_DDS_LOG.CLIENT_SERVER
is '客户端/服务器(CLIENT/SERVER)';
comment on column BGK_DDS_LOG.USER_ID
is '用户id';
comment on column BGK_DDS_LOG.TOTAL_COUNT
is '数据总量';
comment on column BGK_DDS_LOG.OK_COUNT
is '成功条数';
comment on column BGK_DDS_LOG.BEGIN_TIME
is '开始时间';
comment on column BGK_DDS_LOG.END_TIME
is '结束时间 ';
comment on column BGK_DDS_LOG.MEMO
is '备注';
